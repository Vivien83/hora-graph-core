```rust
fn test_entity_creation() {

    let b = hora.add_entity("language", "Rust", None, None).unwrap();
    let fact = hora.add_fact(a, b, "built_with", "hora is built with Rust", None).unwrap();
    let edges = hora.get_entity_facts(a).unwrap();
    assert_eq!(edges.len(), 1);
}
```

**Tests :**
```rust
#[test]
fn test_invalidate_fact() {
    let mut hora = HoraCore::new(HoraConfig::default()).unwrap();
    // ... creer entites + fait
    hora.invalidate_fact(fact_id).unwrap();
    let fact = hora.get_fact(fact_id).unwrap().unwrap();
    assert!(fact.invalid_at > 0);
    // Le fait existe toujours mais est marque comme invalide
}

#[test]
fn test_delete_entity_cascades() {
    // ... creer 2 entites + fait entre elles
    hora.delete_entity(a).unwrap();
    // Le fait doit etre supprime aussi
    assert!(hora.get_fact(fact_id).unwrap().is_none());
}

#[test]
fn test_properties() {
    let id = hora.add_entity("project", "hora",
        Some(props! { "language" => "Rust", "stars" => 42 }), None).unwrap();
    let entity = hora.get_entity(id).unwrap().unwrap();
    assert_eq!(entity.props.get("language"), Some(&PropertyValue::String("Rust".into())));
}
```

## Risques identifies pour v0.1

| Risque | Impact | Mitigation |
|--------|--------|-----------|
| HashMap pas assez rapide pour > 1M entities | Benchmark rate | Switcher vers BTreeMap ou arena allocation |
| Serialisation binaire manuelle = bugs | Data corruption | Property-based tests avec proptest |
| napi-rs v3 breaking changes | Binding casse | Pin la version exacte |
| CSR pas necessaire en v0.1 | Complexite prematuree | Utiliser Vec + HashMap, CSR en v0.2 |
